## Valid Parentheses ( Easy )

```javascript
/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function (s) {
    const stack = [];
    const map = {
        ')': '(',
        '}': '{',
        ']': '[',
    };

    for (let value of s) {
        if (['(', '{', '['].includes(value)) {
            stack.push(value);
        } else {
            if (stack.length === 0 || stack.pop() !== map[value]) {
                return false;
            }
        }
    }

    return stack.length === 0;
};
```
