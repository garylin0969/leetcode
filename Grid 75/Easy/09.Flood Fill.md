## Flood Fill ( Easy )

```javascript
/**
 * @param {number[][]} image
 * @param {number} sr
 * @param {number} sc
 * @param {number} color
 * @return {number[][]}
 */
var floodFill = function (image, sr, sc, color) {
    // 記錄起始像素的原始顏色
    const currentColor = image[sr][sc];

    // 如果起始像素已經是目標顏色，直接返回
    if (currentColor === color) {
        return image;
    }

    // 定義遞迴函數
    function fill(x, y) {
        // 檢查邊界和是否需要填充
        if (x < 0 || x >= image.length || y < 0 || y >= image[0].length || image[x][y] !== currentColor) {
            return;
        }

        // 填充當前像素
        image[x][y] = color;

        // 遞迴處理四個方向
        fill(x - 1, y); // 上
        fill(x + 1, y); // 下
        fill(x, y - 1); // 左
        fill(x, y + 1); // 右
    }

    // 從起始點開始填充
    fill(sr, sc);
    return image;
};
```
